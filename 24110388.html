import math

def mean(data):
    if not data:
        raise ValueError("Data cannot be empty")
    return sum(data) / len(data)

def median(data):
    if not data:
        raise ValueError("Data cannot be empty")
    data = sorted(data)
    n = len(data)
    if n % 2 == 0:
        return (data[n // 2] + data[n // 2 - 1]) / 2
    else:
        return data[n // 2]

def mode(data):
    if not data:
        raise ValueError("Data cannot be empty")
    frequency = {}
    for i in data:
        frequency[i] = frequency.get(i, 0) + 1
    max_count = max(frequency.values())
    modes = [num for num, count in frequency.items() if count == max_count]
    return modes if len(modes) > 1 else modes[0]  # Return single mode or list

def sd(data):
    if not data:
        raise ValueError("Data cannot be empty")
    n = len(data)
    if n < 2:
        raise ValueError("Standard deviation requires at least two data points")
    m = mean(data)
    total = sum((x - m) ** 2 for x in data)
    return math.sqrt(total / (n - 1))

# Test data
data = [1, 2, 3, 4, 5, 2, 2]
print("mean:", mean(data))
print("median:", median(data))
print("mode:", mode(data))
print("standard deviation:", sd(data))

def normal(data):
    mn=min(data)
    mx=max(data)
    n=[]
    for x in data:
        num=(x-mn)/(mx-mn)
        n.append(num)
    return n
